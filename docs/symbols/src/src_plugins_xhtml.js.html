 34 		var base = this;
 35

 37 			indentStr: '\t'

 40 		/**
 41 		 * Array containing the output, used as it's faster
 42 		 * than string concation in slow browsers.
 43 		 * @type {Array}
 44 		 * @private
 45 		 */
 46 		var outputStringBuilder = [];
 47 
 48 		/**
 49 		 * Current indention level
 50 		 * @type {Number}
 51 		 * @private
 52 		 */
 53 		var currentIndent = 0;
 54 
 55 		/**
 56 		 * @private
 57 		 */
 58 		var	escapeEntites,
 59 			trim,
 60 			serializeNode,
 61 			handleDoc,
 62 			handleElement,
 63 			handleCdata,
 64 			handleComment,
 65 			handleText,
 66 			indent,
 67 			canIndent,
 68 			newline;

 70 		/**
 71 		 * Escapes XHTML entities
 72 		 * @param  {String} str
 73 		 * @return {String}
 74 		 * @private
 75 		 */
 76 		escapeEntites = function(str) {
 77 			var entites = {
 78 				'&': '&',
 79 				'<': '<',
 80 				'>': '>',
 81 				'"': '"'
 82 			};
 83 
 84 			return !str ? '' : str.replace(/[&<>"]/g, function(entity) {
 85 				return entites[entity] || entity;
 86 			});
 87 		};
 88 
 89 		/**
 90 		 * @param  {string} str
 91 		 * @return {string}
 92 		 * @private
 93 		 */
 94 		trim = function(str) {
 95 			return str
 96 				// new lines in text nodes are always ignored in normal handling
 97 				.replace(/[\r\n]/, "")
 98 				.replace(/[^\S|\u00A0]+/g, " ");
 99 		};

101 		/**
102 		 * Serializes a node to XHTML
103 		 * @param  {Node}	node		Node to serialize
104 		 * @param  {Boolean}	onlyChildren	If to only serialize the nodes children and not the node its self
105 		 * @return {String}			The serialized node
106 		 * @name serialize
107 		 * @memberOf jQuery.sceditor.XHTMLSerializer.prototype
108 		 * @since v1.4.1
109 		 */
110 		base.serialize = function(node, onlyChildren) {
111 			outputStringBuilder = [];
112 
113 			if(onlyChildren)
114 			{
115 				node = node.firstChild;
116 
117 				while(node)
118 				{
119 					serializeNode(node);
120 					node = node.nextSibling;
121 				}
122 			}
123 			else
124 				serializeNode(node);
125 
126 			return outputStringBuilder.join('');
127 		};
128 
129 		/**
130 		 * Serializes a node to the outputStringBuilder
131 		 * @param  {Node} node
132 		 * @return {Void}
133 		 * @private
134 		 */
135 		serializeNode = function(node) {
136 			switch(node.nodeType)
137 			{
138 				case 1: // element
139 					var tagName = node.nodeName.toLowerCase();
140 
141 					// IE comment
142 					if(tagName === '!')
143 						handleComment(node);
144 
145 
146 					handleElement(node);
147 					break;
148 
149 				case 3: // text
150 					handleText(node);
151 					break;
152 
153 				case 4: // cdata section
154 					handleCdata(node);
155 					break;
156 
157 				case 8: // comment
158 					handleComment(node);
159 					break;
160 
161 				case 9: // document
162 				case 11: // document fragment
163 					handleDoc(node);
164 					break;
165 
166 
167 				// Ignored types
168 				case 2: // attribute
169 				case 5: // entity ref
170 				case 6: // entity
171 				case 7: // processing instruction
172 				case 10: // document type
173 				case 12: // notation
174 					break;
175 			}
176 		};
177 
178 		/**
179 		 * Handles doc node
180 		 * @param  {Node} node
181 		 * @return {void}
182 		 * @private
183 		 */
184 		handleDoc = function(node) {
185 			var	child;
186 
187 			child = node.firstChild;
188 
189 			while(child)
190 			{
191 				serializeNode(child);
192 				child = child.nextSibling;
193 			}
194 		};
195 
196 		/**
197 		 * Handles element nodes
198 		 * @param  {Node} node
199 		 * @return {void}
200 		 * @private
201 		 */
202 		handleElement = function(node) {
203 			var	child, attr,
204 				tagName     = node.nodeName.toLowerCase(),
205 				i           = node.attributes.length,
206 				selfClosing = !node.firstChild && /^(?:area|base|br|col|embed|hr|img|input|link|meta|param)$/.test(tagName);
207 
208 			if(canIndent(node))
209 				indent();
210 
211 			outputStringBuilder.push('<' + tagName);
212 			while(i--)
213 			{
214 				attr = node.attributes[i];
215 				outputStringBuilder.push(" ", attr.name.toLowerCase(), '="', escapeEntites(attr.value), '"');
216 			}
217 			outputStringBuilder.push(selfClosing ? ' />' : '>');
218 
219 			currentIndent++;
220 
221 			child = node.firstChild;
222 			while(child)
223 			{
224 				serializeNode(child);
225 				child = child.nextSibling;
226 			}
227 
228 			currentIndent--;
229 
230 			if(!selfClosing)
231 			{
232 				if(canIndent(node) && node.firstChild)
233 					indent();
234 
235 				outputStringBuilder.push('</', tagName, '>');
236 			}
237 		};
238 
239 		/**
240 		 * Handles CDATA nodes
241 		 * @param  {Node} node
242 		 * @return {void}
243 		 * @private
244 		 */
245 		handleCdata =  function(node) {
246 			indent();
247 
248 			outputStringBuilder.push('<![CDATA[', escapeEntites(node.nodeValue), ']]>');
249 		};
250 
251 		/**
252 		 * Handles comment nodes
253 		 * @param  {Node} node
254 		 * @return {void}
255 		 * @private
256 		 */
257 		handleComment = function(node) {
258 			indent();
259 
260 			outputStringBuilder.push('<!-- ', escapeEntites(node.nodeValue), ' -->');
261 		};
262 
263 		/**
264 		 * Handles test nodes
265 		 * @param  {Node} node
266 		 * @return {void}
267 		 * @private
268 		 */
269 		handleText = function(node) {
270 			var text = trim(node.nodeValue);
271 
272 			if(text)
273 			{
274 				if(canIndent(node))
275 					indent();
276 
277 				outputStringBuilder.push(escapeEntites(text));
278 			}
279 		};
280 
281 		/**
282 		 * Adds indent to the outputStringBuilder
283 		 * @return {void}
284 		 * @private
285 		 */
286 		indent = function() {
287 			var i = currentIndent;
288 
289 			newline();
290 
291 			while(i--)
292 				outputStringBuilder.push(opts.indentStr);
293 		};
294 
295 		/**
296 		 * Checks if should indent the node or not
297 		 * @param  {Node} node
298 		 * @return {boolean}
299 		 * @private
300 		 */
301 		canIndent = function(node) {
302 			if(node.nodeType !== 1)
303 			{
304 				if(node.previousSibling && $.sceditor.dom.isInline(node.previousSibling))
305 					return false;
306 
307 				node = node.parentNode;
308 			}
309 
310 			return !$.sceditor.dom.isInline(node);
311 		};
312 
313 		/**
314 		 * Adds a new line to the outputStringBuilder
315 		 * @return {void}
316 		 * @private
317 		 */
318 		newline = function() {
319 			outputStringBuilder.push('\n');
320 		};
321 	};

509 		/**
510 		 * Converts a tags name to the name specified
511 		 *
512 		 * @param  {Node} elm
513 		 * @param  {String} newtagName
514 		 * @return {Node} The new node
515 		 * @memberOf jQuery.sceditor.plugins.xhtml.prototype
516 		 */
517 		base.convertTagTo = function(elm, newtagName) {
518 			var	child, attr,
519 				i      = elm.attributes.length,
520 				newTag = elm.ownerDocument.createElement(newtagName);
521 
522 			while(i--)
523 			{
524 				attr = elm.attributes[i];
525 				newTag.setAttribute(attr.name, attr.value);
526 			}
527 
528 			while((child = elm.firstChild))
529 				newTag.appendChild(child);
530 
531 			elm.parentNode.replaceChild(newTag, elm);
532 
533 			return newTag;
534 		};

536 		/**
537 		 * [convertNode description]
538 		 * @param  {[type]} tagName [description]
539 		 * @param  {[type]} $node   [description]
540 		 * @return {[type]}         [description]
541 		 * @private
542 		 */
543 		convertNode = function(tagName, $node) {
544 			if(tagConvertersCache[tagName])
545 			{
546 				$.each(tagConvertersCache[tagName], function(idx, converter) {
547 					if(converter.tags[tagName])
548 					{
549 						$.each(converter.tags[tagName], function(attr, values) {
550 							if(typeof $node.attr(attr) === 'undefined')
551 								return;
552 
553 							if(values && $.inArray($node.attr(attr), values) < 0)
554 								return;
555 
556 							converter.conv.call(base, $node[0]);
557 						});
558 					}
559 					else if(converter.conv)
560 						converter.conv.call(base, $node[0]);
561 				});
562 			}
563 		};
564 		/**
565 		 * Converts any tags/attributes to their XHTML equivalents
566 		 * @param  {Node} node
567 		 * @return {Void}
568 		 * @private
569 		 */
570 		convertTags = function(node) {
571 			if(!tagConvertersCache)
572 				return;
573 
574 			$.sceditor.dom.traverse(node, function(node) {
575 				var	$node   = $(node),
576 					tagName = node.nodeName.toLowerCase();
577 
578 				if(!tagConvertersCache)
579 					return;
580 
581 				convertNode('*', $node);
582 				convertNode(tagName, $node);
583 			}, true);
584 		};

586 		/**
587 		 * Removes any tags that are not white listed or if no
588 		 * tags are white listed it will remove any tags that
589 		 * are black listed.
590 		 * @param  {Node} node
591 		 * @return {Void}
592 		 * @private
593 		 */
594 		removetags = function(node) {
595 			$.sceditor.dom.traverse(node, function(node) {
596 				var	tagName        = node.nodeName.toLowerCase(),
597 					allowedtags    = $.sceditor.plugins.xhtml.allowedTags,
598 					disallowedTags = $.sceditor.plugins.xhtml.disallowedTags;
599 
600 				// cdata section
601 				if(node.nodeType === 4)
602 					tagName = '!cdata';
603 				// comment
604 				else if(tagName === '!' || node.nodeType === 8)
605 					tagName = '!comment';
606 
607 				// 3 is text node which do not get filtered
608 				if(allowedtags && allowedtags.length && node.nodeType !== 3)
609 				{
610 					if($.inArray(tagName, allowedtags) < 0 && node.parentNode)
611 						node.parentNode.removeChild(node);
612 				}
613 				else if(disallowedTags && disallowedTags.length && node.nodeType !== 3)
614 				{
615 					if($.inArray(tagName, disallowedTags) > -1 && node.parentNode)
616 						node.parentNode.removeChild(node);
617 				}
618 			});
619 		};

621 		/**
622 		 * Merges two sets of attribute filters into one
623 		 * @param  {Object}
624 		 * @param  {Object}
625 		 * @return {Object}
626 		 * @private
627 		 */
628 		mergeAttribsFilters = function(filtersA, filtersB) {
629 			var ret = {};
630 
631 			if(filtersA)
632 				$.extend(ret, filtersA);
633 
634 			if(!filtersB)
635 				return ret;
636 
637 			$.each(filtersB, function(attrName, values) {
638 
639 				if($.isArray(ret[attrName]))
640 				{
641 					if($.isArray(values))
642 						ret[attrName] = $.merge(ret[attrName], values);
643 				}
644 				else
645 					ret[attrName] = values;
646 			});
647 
648 			return ret;
649 		};

651 		/**
652 		 * Removes any attributes that are not white listed or
653 		 * if no attributes are white listed it will remove
654 		 * any attributes that are black listed.
655 		 * @param  {Node} node
656 		 * @return {Void}
657 		 * @private
658 		 */
659 		removeAttribs = function(node) {
660 			attrsCache = {};
661 
662 			$.sceditor.dom.traverse(node, function(node) {
663 				if(!node.attributes)
664 					return;
665 
666 				var	attr, removeAttr,
667 					tagName           = node.nodeName.toLowerCase(),
668 					allowedAttribs    = $.sceditor.plugins.xhtml.allowedAttribs,
669 					disallowedAttribs = $.sceditor.plugins.xhtml.disallowedAttribs,
670 					attrsLength       = node.attributes.length;
671 
672 				if(allowedAttribs && !$.isEmptyObject(allowedAttribs) && attrsLength)
673 				{
674 					if(!attrsCache[tagName])
675 						attrsCache[tagName] = mergeAttribsFilters(allowedAttribs['*'], allowedAttribs[tagName]);
676 
677 					while(attrsLength--)
678 					{
679 						attr       = node.attributes[attrsLength];
680 						removeAttr = typeof attrsCache[tagName][attr.name] === 'undefined';
681 
682 						if($.isArray(attrsCache[tagName][attr.name]) && $.inArray(attr.value, attrsCache[tagName][attr.name]) < 0)
683 							removeAttr = true;
684 
685 						if(removeAttr)
686 							node.removeAttribute(attr.name);
687 					}
688 				}
689 				else if(disallowedAttribs && !$.isEmptyObject(disallowedAttribs) && attrsLength)
690 				{
691 					if(!attrsCache[tagName])
692 						attrsCache[tagName] = mergeAttribsFilters(disallowedAttribs['*'], disallowedAttribs[tagName]);
693 
694 					while(attrsLength--)
695 					{
696 						attr       = node.attributes[attrsLength];
697 						removeAttr = typeof attrsCache[tagName][attr.name] !== 'undefined';
698 
699 						if($.isArray(attrsCache[tagName][attr.name]) && $.inArray(attr.value, attrsCache[tagName][attr.name]) > -1)
700 							removeAttr = true;
701 
702 						if(removeAttr)
703 							node.removeAttribute(attr.name);
704 					}
705 				}
706 			});
707 		};
708 	};

710 	/**

712 	 * tags that match the criteria.
713 	 * @type {Array}
714 	 * @name jQuery.sceditor.plugins.xhtml.converters
