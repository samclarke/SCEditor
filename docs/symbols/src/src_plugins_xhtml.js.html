 28 	 * XHTMLSerializer part of the XHTML plugin.
 29 	 *
 30 	 * @class XHTMLSerializer
 31 	 * @name jQuery.sceditor.XHTMLSerializer

 34 	$.sceditor.XHTMLSerializer = function() {
 35 		var base = this;
 36 
 37 		var opts = {
 38 			indentStr: '\t'
 39 		};
 40 
 41 		/**
 42 		 * Array containing the output, used as it's faster
 43 		 * than string concation in slow browsers.
 44 		 * @type {Array}
 45 		 * @private
 46 		 */
 47 		var outputStringBuilder = [];
 48 
 49 		/**
 50 		 * Current indention level
 51 		 * @type {Number}
 52 		 * @private
 53 		 */
 54 		var currentIndent = 0;
 55 
 56 		/**
 57 		 * @private
 58 		 */
 59 		var	escapeEntites,
 60 			trim,
 61 			serializeNode,
 62 			handleDoc,
 63 			handleElement,
 64 			handleCdata,
 65 			handleComment,
 66 			handleText,
 67 			indent,
 68 			canIndent,
 69 			newline;

 71 		/**
 72 		 * Escapes XHTML entities
 73 		 * @param  {String} str
 74 		 * @return {String}
 75 		 * @private
 76 		 */
 77 		escapeEntites = function(str) {
 78 			var entites = {
 79 				'&': '&',
 80 				'<': '<',
 81 				'>': '>',
 82 				'"': '"'
 83 			};
 84 
 85 			return !str ? '' : str.replace(/[&<>"]/g, function(entity) {
 86 				return entites[entity] || entity;
 87 			});
 88 		};
 89 
 90 		/**
 91 		 * @param  {string} str
 92 		 * @return {string}
 93 		 * @private
 94 		 */
 95 		trim = function(str) {
 96 			return str
 97 				// new lines in text nodes are always ignored in normal handling
 98 				.replace(/[\r\n]/, "")
 99 				.replace(/[^\S|\u00A0]+/g, " ");
100 		};
101 
102 		/**
103 		 * Serializes a node to XHTML
104 		 * @param  {Node}	node		Node to serialize
105 		 * @param  {Boolean}	onlyChildren	If to only serialize the nodes children and not the node its self
106 		 * @return {String}			The serialized node
107 		 * @name serialize
108 		 * @memberOf jQuery.sceditor.XHTMLSerializer.prototype
109 		 * @since v1.4.1
110 		 */
111 		base.serialize = function(node, onlyChildren) {
112 			outputStringBuilder = [];
113 
114 			if(onlyChildren)
115 			{
116 				node = node.firstChild;
117 
118 				while(node)
119 				{
120 					serializeNode(node);
121 					node = node.nextSibling;
122 				}
123 			}
124 			else
125 				serializeNode(node);
126 
127 			return outputStringBuilder.join('');
128 		};
129 
130 		/**
131 		 * Serializes a node to the outputStringBuilder
132 		 * @param  {Node} node
133 		 * @return {Void}
134 		 * @private
135 		 */
136 		serializeNode = function(node) {
137 			switch(node.nodeType)
138 			{
139 				case 1: // element
140 					var tagName = node.nodeName.toLowerCase();
141 
142 					// IE comment
143 					if(tagName === '!')
144 						handleComment(node);
145 
146 
147 					handleElement(node);
148 					break;
149 
150 				case 3: // text
151 					handleText(node);
152 					break;
153 
154 				case 4: // cdata section
155 					handleCdata(node);
156 					break;
157 
158 				case 8: // comment
159 					handleComment(node);
160 					break;
161 
162 				case 9: // document
163 				case 11: // document fragment
164 					handleDoc(node);
165 					break;
166 
167 
168 				// Ignored types
169 				case 2: // attribute
170 				case 5: // entity ref
171 				case 6: // entity
172 				case 7: // processing instruction
173 				case 10: // document type
174 				case 12: // notation
175 					break;
176 			}
177 		};
178 
179 		/**
180 		 * Handles doc node
181 		 * @param  {Node} node
182 		 * @return {void}
183 		 * @private
184 		 */
185 		handleDoc = function(node) {
186 			var	child;
187 
188 			child = node.firstChild;
189 
190 			while(child)
191 			{
192 				serializeNode(child);
193 				child = child.nextSibling;
194 			}
195 		};
196 
197 		/**
198 		 * Handles element nodes
199 		 * @param  {Node} node
200 		 * @return {void}
201 		 * @private
202 		 */
203 		handleElement = function(node) {
204 			var	child, attr,
205 				tagName     = node.nodeName.toLowerCase(),
206 				i           = node.attributes.length,
207 				selfClosing = !node.firstChild && /^(?:area|base|br|col|embed|hr|img|input|link|meta|param)$/.test(tagName);
208 
209 			if(canIndent(node))
210 				indent();
211 
212 			outputStringBuilder.push('<' + tagName);
213 			while(i--)
214 			{
215 				attr = node.attributes[i];
216 				outputStringBuilder.push(" ", attr.name.toLowerCase(), '="', escapeEntites(attr.value), '"');
217 			}
218 			outputStringBuilder.push(selfClosing ? ' />' : '>');
219 
220 			currentIndent++;
221 
222 			child = node.firstChild;
223 			while(child)
224 			{
225 				serializeNode(child);
226 				child = child.nextSibling;
227 			}
228 
229 			currentIndent--;
230 
231 			if(!selfClosing)
232 			{
233 				if(canIndent(node) && node.firstChild)
234 					indent();
235 
236 				outputStringBuilder.push('</', tagName, '>');
237 			}
238 		};
239 
240 		/**
241 		 * Handles CDATA nodes
242 		 * @param  {Node} node
243 		 * @return {void}
244 		 * @private
245 		 */
246 		handleCdata =  function(node) {
247 			indent();
248 
249 			outputStringBuilder.push('<![CDATA[', escapeEntites(node.nodeValue), ']]>');
250 		};
251 
252 		/**
253 		 * Handles comment nodes
254 		 * @param  {Node} node
255 		 * @return {void}
256 		 * @private
257 		 */
258 		handleComment = function(node) {
259 			indent();
260 
261 			outputStringBuilder.push('<!-- ', escapeEntites(node.nodeValue), ' -->');
262 		};
263 
264 		/**
265 		 * Handles test nodes
266 		 * @param  {Node} node
267 		 * @return {void}
268 		 * @private
269 		 */
270 		handleText = function(node) {
271 			var text = trim(node.nodeValue);
272 
273 			if(text)
274 			{
275 				if(canIndent(node))
276 					indent();
277 
278 				outputStringBuilder.push(escapeEntites(text));
279 			}
280 		};
281 
282 		/**
283 		 * Adds indent to the outputStringBuilder
284 		 * @return {void}
285 		 * @private
286 		 */
287 		indent = function() {
288 			var i = currentIndent;
289 
290 			newline();
291 
292 			while(i--)
293 				outputStringBuilder.push(opts.indentStr);
294 		};
295 
296 		/**
297 		 * Checks if should indent the node or not
298 		 * @param  {Node} node
299 		 * @return {boolean}
300 		 * @private
301 		 */
302 		canIndent = function(node) {
303 			if(node.nodeType !== 1)
304 			{
305 				if(node.previousSibling && $.sceditor.dom.isInline(node.previousSibling))
306 					return false;
307 
308 				node = node.parentNode;
309 			}
310 
311 			// first child of a block element
312 			if(!node.previousSibling && !$.sceditor.dom.isInline(node.parentNode))
313 				return true;
314 
315 			return !$.sceditor.dom.isInline(node);
316 		};
317 
318 		/**
319 		 * Adds a new line to the outputStringBuilder
320 		 * @return {void}
321 		 * @private
322 		 */
323 		newline = function() {
324 			// Don't add a new line if it's the first element
325 			if(outputStringBuilder.length)
326 				outputStringBuilder.push('\n');
327 		};
328 	};

517 		 * Converts a tags name to the name specified
518 		 *
519 		 * @param  {Node} elm

521 		 * @return {Node} The new node
522 		 * @memberOf jQuery.sceditor.plugins.xhtml.prototype
523 		 */
524 		base.convertTagTo = function(elm, newtagName) {
525 			var	child, attr,
526 				i      = elm.attributes.length,
527 				newTag = elm.ownerDocument.createElement(newtagName);
528 
529 			while(i--)
530 			{

532 				newTag.setAttribute(attr.name, attr.value);
533 			}
534 
535 			while((child = elm.firstChild))

538 			elm.parentNode.replaceChild(newTag, elm);
539

543 		/**
544 		 * [convertNode description]
545 		 * @param  {[type]} tagName [description]
546 		 * @param  {[type]} $node   [description]
547 		 * @return {[type]}         [description]
548 		 * @private
549 		 */
550 		convertNode = function(tagName, $node) {
551 			if(tagConvertersCache[tagName])
552 			{
553 				$.each(tagConvertersCache[tagName], function(idx, converter) {
554 					if(converter.tags[tagName])
555 					{
556 						$.each(converter.tags[tagName], function(attr, values) {
557 							if(typeof $node.attr(attr) === 'undefined')
558 								return;
559 
560 							if(values && $.inArray($node.attr(attr), values) < 0)
561 								return;
562 
563 							converter.conv.call(base, $node[0]);
564 						});
565 					}
566 					else if(converter.conv)
567 						converter.conv.call(base, $node[0]);
568 				});
569 			}
570 		};
571 		/**
572 		 * Converts any tags/attributes to their XHTML equivalents
573 		 * @param  {Node} node
574 		 * @return {Void}
575 		 * @private
576 		 */
577 		convertTags = function(node) {
578 			if(!tagConvertersCache)
579 				return;
580 
581 			$.sceditor.dom.traverse(node, function(node) {
582 				var	$node   = $(node),
583 					tagName = node.nodeName.toLowerCase();
584 
585 				if(!tagConvertersCache)
586 					return;
587 
588 				convertNode('*', $node);
589 				convertNode(tagName, $node);
590 			}, true);
591 		};

593 		/**
594 		 * Removes any tags that are not white listed or if no
595 		 * tags are white listed it will remove any tags that
596 		 * are black listed.
597 		 * @param  {Node} node
598 		 * @return {Void}
599 		 * @private
600 		 */
601 		removetags = function(node) {
602 			$.sceditor.dom.traverse(node, function(node) {
603 				var	tagName        = node.nodeName.toLowerCase(),
604 					allowedtags    = $.sceditor.plugins.xhtml.allowedTags,
605 					disallowedTags = $.sceditor.plugins.xhtml.disallowedTags;
606 
607 				// cdata section
608 				if(node.nodeType === 4)
609 					tagName = '!cdata';
610 				// comment
611 				else if(tagName === '!' || node.nodeType === 8)
612 					tagName = '!comment';
613 
614 				// 3 is text node which do not get filtered
615 				if(allowedtags && allowedtags.length && node.nodeType !== 3)
616 				{
617 					if($.inArray(tagName, allowedtags) < 0 && node.parentNode)
618 						node.parentNode.removeChild(node);
619 				}
620 				else if(disallowedTags && disallowedTags.length && node.nodeType !== 3)
621 				{
622 					if($.inArray(tagName, disallowedTags) > -1 && node.parentNode)
623 						node.parentNode.removeChild(node);
624 				}
625 			});
626 		};

628 		/**
629 		 * Merges two sets of attribute filters into one
630 		 * @param  {Object}
631 		 * @param  {Object}
632 		 * @return {Object}
633 		 * @private
634 		 */
635 		mergeAttribsFilters = function(filtersA, filtersB) {
636 			var ret = {};
637 
638 			if(filtersA)
639 				$.extend(ret, filtersA);
640 
641 			if(!filtersB)
642 				return ret;
643 
644 			$.each(filtersB, function(attrName, values) {
645 
646 				if($.isArray(ret[attrName]))
647 				{
648 					if($.isArray(values))
649 						ret[attrName] = $.merge(ret[attrName], values);
650 				}
651 				else
652 					ret[attrName] = values;
653 			});
654 
655 			return ret;
656 		};
657 
658 		/**
659 		 * Removes any attributes that are not white listed or
660 		 * if no attributes are white listed it will remove
661 		 * any attributes that are black listed.
662 		 * @param  {Node} node
663 		 * @return {Void}
664 		 * @private
665 		 */
666 		removeAttribs = function(node) {
667 			attrsCache = {};
668 
669 			$.sceditor.dom.traverse(node, function(node) {
670 				if(!node.attributes)
671 					return;
672 
673 				var	attr, removeAttr,
674 					tagName           = node.nodeName.toLowerCase(),
675 					allowedAttribs    = $.sceditor.plugins.xhtml.allowedAttribs,
676 					disallowedAttribs = $.sceditor.plugins.xhtml.disallowedAttribs,
677 					attrsLength       = node.attributes.length;
678 
679 				if(allowedAttribs && !$.isEmptyObject(allowedAttribs) && attrsLength)
680 				{
681 					if(!attrsCache[tagName])
682 						attrsCache[tagName] = mergeAttribsFilters(allowedAttribs['*'], allowedAttribs[tagName]);
683 
684 					while(attrsLength--)
685 					{
686 						attr       = node.attributes[attrsLength];
687 						removeAttr = typeof attrsCache[tagName][attr.name] === 'undefined';
688 
689 						if($.isArray(attrsCache[tagName][attr.name]) && $.inArray(attr.value, attrsCache[tagName][attr.name]) < 0)
690 							removeAttr = true;
691 
692 						if(removeAttr)
693 							node.removeAttribute(attr.name);
694 					}
695 				}
696 				else if(disallowedAttribs && !$.isEmptyObject(disallowedAttribs) && attrsLength)
697 				{
698 					if(!attrsCache[tagName])
699 						attrsCache[tagName] = mergeAttribsFilters(disallowedAttribs['*'], disallowedAttribs[tagName]);
700 
701 					while(attrsLength--)
702 					{
703 						attr       = node.attributes[attrsLength];
704 						removeAttr = typeof attrsCache[tagName][attr.name] !== 'undefined';
705 // this should check if only certian values so should be === null?
706 						if($.isArray(attrsCache[tagName][attr.name]) && $.inArray(attr.value, attrsCache[tagName][attr.name]) > -1)
