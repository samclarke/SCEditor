121 			 * @param  {TokenizeToken|Int} splitAt The child to split at or the index of the child
122 			 * @return {TokenizeToken} The right half of the split token or null if failed
123 			 */
124 			splitAt: function(splitAt) {
125 				var	clone,
126 					base          = this,
127 					splitAtLength = 0,
128 					i             = base.children.length;
129 
130 				if(typeof object !== "number")
131 					splitAt = $.inArray(splitAt, base.children);

135 
136 				// Work out how many items are on the right side of the split
137 				// to pass to splice()
138 				while(i--)
139 				{
140 					if(i >= splitAt)

142 					else
143 						i = false;
144 				}
145 
146 				clone          = base.clone();
147 				clone.children = base.children.splice(splitAt, splitAtLength);
148 				return clone;
149 			}

158 		/**
159 		 * Takes a BBCode string and splits it into open, content and close tags.
160 		 *
161 		 * It does no checking to verify a tag has a matching open or closing tag
162 		 * or if the tag is valid child of any tag before it. For that the tokens
163 		 * should be passed to the parse function.
164 		 *
165 		 * @param {String} str
166 		 * @return {Array}
167 		 * @memberOf jQuery.sceditor.BBCodeParser.prototype
168 		 */
169 		base.tokenize = function(str) {
170 			var	matches, type, i,
171 				toks   = [],
172 				tokens = [
173 					// Close must come before open as they are
174 					// the same except close has a / at the start.
175 					{
176 						type: 'close',
177 						regex: /^\[\/[^\[\]]+\]/
178 					},
179 					{
180 						type: 'open',
181 						regex: /^\[[^\[\]]+\]/
182 					},
183 					{
184 						type: 'newline',
185 						regex: /^(\r\n|\r|\n)/
186 					},
187 					{
188 						type: 'content',
189 						regex: /^([^\[\r\n]+|\[)/
190 					}
191 				];
192 
193 			tokens.reverse();
194 
195 			strloop:
196 			while(str.length)
197 			{
198 				i = tokens.length;
199 				while(i--)
200 				{
201 					type = tokens[i].type;
202 
203 					// Check if the string matches any of the tokens
204 					if(!(matches = str.match(tokens[i].regex)) || !matches[0])
205 						continue;
206 
207 					// Add the match to the tokens list
208 					toks.push(tokenizeTag(type, matches[0]));
209 
210 					// Remove the match from the string
211 					str = str.substr(matches[0].length);
212 
213 					// The token has been added so start again
214 					continue strloop;
215 				}
216 
217 				// If there is anything left in the string which doesn't match
218 				// any of the tokens then just assume it's content and add it.
219 				if(str.length)
220 					toks.push(tokenizeTag(tokenType.content, str));
221 
222 				str = "";
223 			}
224 
225 			return toks;
226 		};

228 		/**
229 		 * Extracts the name an params from a tag
230 		 *
231 		 * @param {Object} token
232 		 * @return {Object}
233 		 * @private
234 		 */
235 		tokenizeTag = function(type, val) {
236 			var matches, attrs, name;
237 
238 			// Extract the name and attributes from opening tags and
239 			// just the name from closing tags.
240 			if(type === "open")
241 			{
242 				matches = val.match(/\[([^\]\s=]+)(?:([^\]]+))?\]/);
243 
244 				// Make sure there was a match otherwise it must be content
245 				if(matches)
246 				{
247 					name = lower(matches[1]);
248 
249 					if(matches[2] && (matches[2] = $.trim(matches[2])))
250 						attrs = tokenizeAttrs(matches[2]);
251 				}
252 				else
253 				{
254 					type = 'content';
255 					name = '#';
256 				}
257 			}
258 			else if(type === "close")
259 			{
260 				matches = val.match(/\[\/([^\[\]]+)\]/);
261 
262 				if(!matches)
263 				{
264 					type = 'content';
265 					name = '#';
266 				}
267 				else
268 					name = lower(matches[1]);
269 			}
270 			else if(type === "newline")
271 				name = '#newline';
272 			else
273 				name = '#';
274 
275 			return new TokenizeToken(type, name, val, attrs);
276 		};

278 		/**
279 		 * Extracts the individual attributes from a string containing
280 		 * all the attributes.
281 		 *
282 		 * @param {String} attrs
283 		 * @return {Array} Assoc array of attributes
284 		 * @private
285 		 */
286 		tokenizeAttrs = function(attrs) {
287 			var	matches, unquote,
288 				/*
289 				([^\s=]+)					Anything that's not a space or equals
290 				=						Equals =
291 				(?:
292 					(?:
293 						(["'])				The opening quote
294 						(
295 							(?:\\\2|[^\2])*?	Anything that isn't the unescaped opening quote
296 						)
297 						\2				The opening quote again which will now close the string
298 					)
299 						|				If not a quoted string then match
300 					(
301 						(?:.(?!\s\S+=))*.?		Anything that isn't part of [space][non-space][=] which would be a new attribute
302 					)
303 				)
304 				 */
305 				atribsRegex = /([^\s=]+)=(?:(?:(["'])((?:\\\2|[^\2])*?)\2)|((?:.(?!\s\S+=))*.))/g,
306 				ret         = {};
307 
308 			//
309 			unquote = function(str, quote) {
310 				if(quote)
311 					str = str.replace('\\\\', '\\').replace('\\' + quote, quote);
312 
313 				return str;
314 			};
315 
316 			// if only one attribute then remove the = from the start and strip any quotes
317 			if(attrs.charAt(0) === "=" && attrs.split("=").length <= 2)
318 				ret.defaultattr = $.sceditorBBCodePlugin.stripQuotes(attrs.substr(1));
319 			else
320 			{
321 				if(attrs.charAt(0) === "=")
322 					attrs = "defaultattr" + attrs;
323 
324 				// No need to strip quotes here, the regex will do that.
325 				while((matches = atribsRegex.exec(attrs)))
326 					ret[lower(matches[1])] = unquote(matches[3], matches[2]) || matches[4];
327 			}
328 
329 			return ret;
330 		};

405 		// TODO: Tidy this parseTokens() function up a bit.
406 		/**
407 		 * Parses an array of tokens created by tokenize()
408 		 *
409 		 * @param  {Array} toks
410 		 * @return {Array} Parsed tokens
411 		 * @see tokenize()
412 		 * @private
413 		 */
414 		parseTokens = function(toks) {
415 			var	token, bbcode, curTok, clone, i, previous, next,
416 				cloned     = [],
417 				output     = [],
418 				openTags   = [],
419 				/**
420 				 * Returns the currently open tag or undefined
421 				 * @return {TokenizeToken}
422 				 */
423 				currentOpenTag = function() {
424 					return last(openTags);
425 				},
426 				/**
427 				 * Adds a tag to either the current tags children
428 				 * or to the output array.
429 				 * @param {TokenizeToken} token
430 				 * @private
431 				 */
432 				addTag = function(token) {
433 					if(currentOpenTag())
434 						currentOpenTag().children.push(token);
435 					else
436 						output.push(token);
437 				},
438 				/**
439 				 * Checks if this tag closes the current tag
440 				 * @param  {String} name
441 				 * @return {Void}
442 				 */
443 				closesCurrentTag = function(name) {
444 					return currentOpenTag() &&
445 						(bbcode = base.bbcodes[currentOpenTag().name]) &&
446 						bbcode.closedBy &&
447 						$.inArray(name, bbcode.closedBy) > -1;
448 				};
449 
450 			while((token = toks.shift()))
451 			{
452 				next = toks[0];
453 
454 				switch(token.type)
455 				{
456 					case tokenType.open:
457 						// Check it this closes a parent, i.e. for lists [*]one [*]two
458 						if(closesCurrentTag(token.name))
459 							openTags.pop();
460 
461 						addTag(token);
462 						bbcode = base.bbcodes[token.name];
463 
464 						// If this tag is not self closing and it has a closing tag then it is open and has children so
465 						// add it to the list of open tags. If it is a valid BBCode, e.g. in the BBCode list but is missing
466 						// an end tag then just assume it's misisng and include everything as it's children.
467 						if((!bbcode || !bbcode.isSelfClosing) && (bbcode || hasTag(token.name, tokenType.close, toks)))
468 							openTags.push(token);
469 
470 						break;
471 
472 					case tokenType.close:
473 						// check if this closes the current tag, e.g. [/list] would close an open [*]
474 						if(currentOpenTag() && token.name !== currentOpenTag().name && closesCurrentTag('/' + token.name))
475 							openTags.pop();
476 
477 						// If this is closing the currently open tag just pop thfcloe
478 						// tage off the open tags array
479 						if(currentOpenTag() && token.name === currentOpenTag().name)
480 						{
481 							currentOpenTag().closing = token;
482 							openTags.pop();
483 						}
484 						// If this is closing an open tag that is the parent of the current
485 						// tag then clone all the tags including the current one until
486 						// reaching the parent that is being closed. Close the parent and then
487 						// add the clones back in.
488 						else if(hasTag(token.name, tokenType.open, openTags))
489 						{
490 							// Remove the tag from the open tags
491 							while((curTok = openTags.pop()))
492 							{
493 								// If it's the tag that is being closed then
494 								// discard it and break the loop.
495 								if(curTok.name === token.name)
496 								{
497 									curTok.closing = token;
498 									break;
499 								}
500 
501 								// Otherwise clone this tag and then add any
502 								// previously cloned tags as it's children
503 								clone = curTok.clone();
504 
505 								if(cloned.length > 1)
506 									clone.children.push(last(cloned));
507 
508 								cloned.push(clone);
509 							}
510 
511 							// Add the last cloned child to the now current tag
512 							// (the parent of the tag which was being closed)
513 							addTag(last(cloned));
514 
515 							// Add all the cloned tags to the open tags list
516 							i = cloned.length;
517 							while(i--)
518 								openTags.push(cloned[i]);
519 
520 							cloned.length = 0;
521 						}
522 						// This tag is closing nothing so treat it as content
523 						else
524 						{
525 							token.type = tokenType.content;
526 							addTag(token);
527 						}
528 						break;
529 
530 					case tokenType.newline:
531 						// handle things like
532 						//     [*]list\nitem\n[*]list1
533 						// where it should come out as
534 						//     [*]list\nitem[/*]\n[*]list1[/*]
535 						// instead of
536 						//     [*]list\nitem\n[/*][*]list1[/*]
537 						if(currentOpenTag() && next && closesCurrentTag((next.type === tokenType.close ? '/' : '') + next.name))
538 						{
539 							// skip if the next tag is the closing tag for the option tag, i.e. [/*]
540 							if(!(next.type === tokenType.close && next.name === currentOpenTag().name))
541 							{
542 								bbcode = base.bbcodes[currentOpenTag().name];
543 
544 								if(bbcode && bbcode.breakAfter)
545 									openTags.pop();
546 								else if(bbcode && bbcode.isInline === false && base.opts.breakAfterBlock && bbcode.breakAfter !== false)
547 									openTags.pop();
548 							}
549 						}
550 
551 						addTag(token);
552 						break;
553 
554 					default: // content
555 						addTag(token);
556 						break;
557 				}
558 
559 				previous = token;
560 			}
561 
562 			return output;
563 		};

565 		/**
566 		 * Normalise all new lines
567 		 *
568 		 * Removes any formatting new lines from the BBCode
569 		 * leaving only content ones. I.e. for a list:
570 		 *
571 		 * [list]
572 		 * [*] list item one
573 		 * with a line break
574 		 * [*] list item two
575 		 * [/list]
576 		 *
577 		 * would become
578 		 *
579 		 * [list] [*] list item one
580 		 * with a line break [*] list item two [/list]
581 		 *
582 		 * Which makes it easier to convert to HTML or add
583 		 * the formmating new lines back in when converting
584 		 * back to BBCode
585 		 *
586 		 * @param  {Array} children
587 		 * @param  {TokenizeToken} parent
588 		 * @param  {Bool} onlyRemoveBreakAfter
589 		 * @return {void}
590 		 */
591 		normaliseNewLines = function(children, parent, onlyRemoveBreakAfter) {
592 			var	token, left, right, bbcode, leftBBCode, rightBBCode,
593 				removedBreakEnd, removedBreakBefore, remove,
594 				childrenLength = children.length,
595 				i              = childrenLength;
596 
597 			if(parent)
598 				bbcode = base.bbcodes[parent.name];
599 
600 			while(i--)
601 			{
602 				if(!(token = children[i]))
603 					continue;
604 
605 				if(token.type === tokenType.newline)
606 				{
607 					left   = i > 0 ? children[i - 1] : null;
608 					right  = i < childrenLength - 1 ? children[i+1] : null;
609 					remove = false;
610 
611 					// Handle the start and end new lines e.g. [tag]\n and \n[/tag]
612 					if(!onlyRemoveBreakAfter && bbcode && bbcode.isSelfClosing !== true)
613 					{
614 						// First child of parent so must be opening line break (breakStartBlock, breakStart) e.g. [tag]\n
615 						if(i === 0)
616 						{
617 							if(bbcode.isInline === false && base.opts.breakStartBlock && bbcode.breakStart !== false)
618 								remove = true;
619 
620 							if(bbcode.breakStart)
621 								remove = true;
622 						}
623 						// Last child of parent so must be end line break (breakEndBlock, breakEnd) e.g. \n[/tag]
624 						// remove last line break (breakEndBlock, breakEnd)
625 						else if (!removedBreakEnd && childrenLength - 1 === i)
626 						{
627 							if(bbcode.isInline === false && base.opts.breakEndBlock && bbcode.breakEnd !== false)
628 								remove = true;
629 
630 							if(bbcode.breakEnd)
631 								remove = true;
632 
633 							removedBreakEnd = remove;
634 						}
635 					}
636 
637 					if(left && left.type === tokenType.open)
638 					{
639 						if((leftBBCode = base.bbcodes[left.name]))
640 						{
641 							if(!onlyRemoveBreakAfter)
642 							{
643 								if(leftBBCode.isInline === false && base.opts.breakAfterBlock && leftBBCode.breakAfter !== false)
644 									remove = true;
645 
646 								if(leftBBCode.breakAfter)
647 									remove = true;
648 							}
649 							else if(leftBBCode.isInline === false)
650 								remove = true;
651 						}
652 					}
653 
654 					if(!onlyRemoveBreakAfter && !removedBreakBefore && right && right.type === tokenType.open)
655 					{
656 						if((rightBBCode = base.bbcodes[right.name]))
657 						{
658 							if(rightBBCode.isInline === false && base.opts.breakBeforeBlock && rightBBCode.breakBefore !== false)
659 								remove = true;
660 
661 							if(rightBBCode.breakBefore)
662 								remove = true;
663 
664 							removedBreakBefore = remove;
665 
666 							if(remove)
667 							{
668 								children.splice(i, 1);
669 								continue;
670 							}
671 						}
672 					}
673 
674 					if(remove)
675 						children.splice(i, 1);
676 
677 					// reset double removedBreakBefore removal protection.
678 					// This is needed for cases like \n\n[\tag] where
679 					// only 1 \n should be removed but without this they both
680 					// would be.
681 					removedBreakBefore = false;
682 				}
683 				else if(token.type === tokenType.open)
684 					normaliseNewLines(token.children, token, onlyRemoveBreakAfter);
685 			}
686 		};

688 		/**
689 		 * Fixes any invalid nesting.
690 		 *
691 		 * If it is a block level element inside 1 or more inline elements
692 		 * then those inline elements will be split at the point where the
693 		 * block level is and the block level element placed between the split
694 		 * parts. i.e.
695 		 *     [inline]textA[blocklevel]textB[/blocklevel]textC[/inline]
696 		 * Will become:
697 		 *     [inline]textA[/inline][blocklevel]textB[/blocklevel][inline]textC[/inline]
698 		 *
699 		 * @param {Array} children
700 		 * @param {Array} [parents] Null if there is no parents
701 		 * @param {Array} [insideInline] Boolean, if insdie an inline element
702 		 * @param {Array} [rootArr] Root array if there is one
703 		 * @return {Array}
704 		 * @private
705 		 */
706 		fixNesting = function(children, parents, insideInline, rootArr) {
707 			var	token, i, parent, parentIndex, parentParentChildren, right,
708 				isInline = function(token) {
709 					var bbcode = base.bbcodes[token.name];
710 
711 					return !bbcode || bbcode.isInline !== false;
712 				};
713 
714 			parents = parents || [];
715 			rootArr = rootArr || children;
716 
717 			// this must check length each time as the length
718 			// can change as tokens are moved around to fix the nesting.
719 			for(i=0; i<children.length; i++)
720 			{
721 				if(!(token = children[i]) || token.type !== tokenType.open)
722 					continue;
723 
724 				if(!isInline(token) && insideInline)
725 				{
726 					// if this is a blocklevel element inside an inline one then split
727 					// the parent at the block level element
728 					parent               = last(parents);
729 					right                = parent.splitAt(token);
730 					parentParentChildren = parents.length > 1 ? parents[parents.length - 2].children : rootArr;
731 
732 					if((parentIndex = $.inArray(parent, parentParentChildren)) > -1)
733 					{
734 						// remove the block level token from the right side of the split
735 						// inlnie element
736 						right.children.splice($.inArray(token, right.children), 1);
737 
738 						// insert the block level token and the right side after the left
739 						// side of the inline token
740 						parentParentChildren.splice(parentIndex+1, 0, token, right);
741 
742 						// return to parents loop as the children have now increased
743 						return;
744 					}
745 
746 				}
747 
748 				parents.push(token);
749 				fixNesting(token.children, parents, insideInline || isInline(token), rootArr);
750 				parents.pop(token);
751 			}
752 		};

754 		/**
755 		 * Fixes any invalid children.
756 		 *
757 		 * If it is an element which isn't allowed as a child of it's parent
758 		 * then it will be converted to content of the parent element. i.e.
759 		 *     [code]Code [b]only[/b] allows text.[/code]
760 		 * Will become:
761 		 *     <code>Code [b]only[/b] allows text.</code>
762 		 * Instead of:
763 		 *     <code>Code <b>only</b> allows text.</code>
764 		 *
765 		 * @param {Array} children
766 		 * @param {Array} [parent] Null if there is no parents
767 		 * @private
768 		 */
769 		fixChildren = function(children, parent) {
770 			var	token, args,
771 				i = children.length;
772 
773 			while(i--)
774 			{
775 				if(!(token = children[i]))
776 					continue;
777 
778 				if(!isChildAllowed(parent, token))
779 				{
780 					// if it is not then convert it to text and see if it
781 					// is allowed
782 					token.name = null;
783 					token.type = tokenType.content;
784 
785 					if(isChildAllowed(parent, token))
786 					{
787 						args = [i+1, 0].concat(token.children);
788 
789 						if(token.closing)
790 						{
791 							token.closing.name = null;
792 							token.closing.type = tokenType.content;
793 							args.push(token.closing);
794 						}
795 
796 						i += args.length - 1;
797 						Array.prototype.splice.apply(children, args);
798 					}
799 					else
800 						parent.children.splice(i, 1);
801 				}
802 
803 				if(token.type === tokenType.open)
804 					fixChildren(token.children, token);
805 			}
806 		};

846 		convertToHTML = function(tokens, isRoot) {
847 			var	token, bbcode, content, html, needsBlockWrap,
848 				blockWrapOpen, isInline, addLineBreak,
849 				ret = [];
850 
851 			while(tokens.length > 0)
852 			{
853 				if(!(token = tokens.shift()))
854 					continue;
855 
856 				if(token.type === tokenType.open)
857 				{
858 					bbcode         = base.bbcodes[token.name];
859 					isInline       = !bbcode || (typeof bbcode.isHtmlInline !== "undefined" ? bbcode.isHtmlInline : bbcode.isInline);
860 					needsBlockWrap = isRoot && (!bbcode || isInline !== false);
861 					content        = convertToHTML(token.children, false);
862 
863 					if(bbcode && bbcode.html)
864 					{
865 						addLineBreak = isInline === false && !bbcode.isPreFormatted && !bbcode.skipLastLineBreak;
866 
867 						// Add placeholder br to end of block level elements in all browsers apart from IE < 9 which
868 						// handle new lines diffrently and don't need one.
869 						if(addLineBreak && !$.sceditor.ie)
870 							content += "<br />";
871 
872 						if($.isFunction(bbcode.html))
873 							html = bbcode.html.call(base, token, token.attrs, content);
874 						else
875 							html = $.sceditorBBCodePlugin.formatString(bbcode.html, content);
876 					}
877 					else
878 						html = token.val + content + (token.closing ? token.closing.val : '');
879 				}
880 				else if(token.type === tokenType.newline)
881 				{
882 					if(!isRoot)
883 					{
884 						ret.push('<br />');
885 						continue;
886 					}
887 
888 					// if already in a block wrap then just closing it
889 					// will act as a line-break.
890 					if(blockWrapOpen)
891 					{
892 						ret.push('</div>\n');
893 						blockWrapOpen = false;
894 						continue;
895 					}
896 
897 					ret.push('<div>');
898 
899 					// Putting BR in a div in IE9 causes it to do a double line break.
900 					if(!$.sceditor.ie)
901 						ret.push('<br />');
902 
903 					// If it's an empty DIV and compatibility mode is below IE8 then
904 					// we must add a non-breaking space to the div otherwise the div
905 					// will be collapsed. Adding a BR works but when you press enter
906 					// to make a newline it suddenly goes back to the normal IE div
907 					// behaviour and creates two lines, one for the newline and one
908 					// for the BR. I'm sure there must be a better fix but I've yet to
909 					// find one.
910 					// Cannot do zoom: 1; or set a height on the div to fix it as that
911 					// causes resize handles to be added to the div when it's clicked on/
912 					if((document.documentMode && document.documentMode < 8) || $.sceditor.ie < 8)
913 						ret.push('\u00a0');
914 
915 					ret.push('</div>\n');
916 					continue;
917 				}
918 				else // content
919 				{
920 					needsBlockWrap = isRoot;
921 					html           = $.sceditor.escapeEntities(token.val);
922 				}
923 
924 				if(needsBlockWrap && !blockWrapOpen)
925 				{
926 					ret.push('<div>');
927 					blockWrapOpen = true;
928 				}
929 				else if(!needsBlockWrap && blockWrapOpen)
930 				{
931 					ret.push('</div>\n');
932 					blockWrapOpen = false;
933 				}
934 
935 				ret.push(html);
936 			}
937 
938 			if(blockWrapOpen)
939 				ret.push('</div>\n');
940 
941 			return ret.join('');
942 		};

959 		 * Converts parsed tokens back into BBCode with the
960 		 * formatting specified in the options and with any
961 		 * fixes specified.
962 		 *
963 		 * @param  {Array} toks Array of parsed tokens from base.parse()
964 		 * @return {String}
965 		 * @private
966 		 */
967 		convertToBBCode = function(toks) {
968 			var	token, attr, bbcode, isBlock, isSelfClosing, quoteType,
969 				breakBefore, breakStart, breakEnd, breakAfter,
970 				// Create an array of strings which are joined together
971 				// before being returned as this is faster in slow browsers.
972 				// (Old versions of IE).
973 				ret = [];
974 
975 			while(toks.length > 0)
976 			{
977 				if(!(token = toks.shift()))
978 					continue;
979 
980 				bbcode        = base.bbcodes[token.name];
981 				isBlock       = !(!bbcode || bbcode.isInline !== false);
982 				isSelfClosing = bbcode && bbcode.isSelfClosing;
983 				breakBefore   = ((isBlock && base.opts.breakBeforeBlock && bbcode.breakBefore !== false) || (bbcode && bbcode.breakBefore));
984 				breakStart    = ((isBlock && !isSelfClosing && base.opts.breakStartBlock && bbcode.breakStart !== false) || (bbcode && bbcode.breakStart));
985 				breakEnd      = ((isBlock && base.opts.breakEndBlock && bbcode.breakEnd !== false) || (bbcode && bbcode.breakEnd));
986 				breakAfter    = ((isBlock && base.opts.breakAfterBlock && bbcode.breakAfter !== false) || (bbcode && bbcode.breakAfter));
987 				quoteType     = (bbcode ? bbcode.quoteType : null) || base.opts.quoteType || $.sceditor.BBCodeParser.QuoteType.auto;
988 
989 				if(!bbcode && token.type === tokenType.open)
990 				{
991 					ret.push(token.val);
992 
993 					if(token.children)
994 						ret.push(convertToBBCode(token.children));
995 
996 					if(token.closing)
997 						ret.push(token.closing.val);
998 				}
999 				else if(token.type === tokenType.open)
1000 				{
1001 					if(breakBefore)
1002 						ret.push('\n');
1003 
1004 					// Convert the tag and it's attributes to BBCode
1005 					ret.push('[' + token.name);
1006 					if(token.attrs)
1007 					{
1008 						if(token.attrs.defaultattr)
1009 						{
1010 							ret.push('=' + quote(token.attrs.defaultattr, quoteType, 'defaultattr'));
1011 							delete token.attrs.defaultattr;
1012 						}
1013 
1014 						for(attr in token.attrs)
1015 							if(token.attrs.hasOwnProperty(attr))
1016 								ret.push(' ' + attr + '=' + quote(token.attrs[attr], quoteType, attr));
1017 					}
1018 					ret.push(']');
1019 
1020 					if(breakStart)
1021 						ret.push('\n');
1022 
1023 					// Convert the tags children to BBCode
1024 					if(token.children)
1025 						ret.push(convertToBBCode(token.children));
1026 
1027 					// add closing tag if not self closing
1028 					if(!isSelfClosing && !bbcode.excludeClosing)
1029 					{
1030 						if(breakEnd)
1031 							ret.push('\n');
1032 
1033 						ret.push('[/' + token.name + ']');
1034 					}
1035 
1036 					if(breakAfter)
1037 						ret.push('\n');
1038 
1039 					// preserve whatever was recognised as the closing tag if
1040 					// it is a self closing tag
1041 					if(token.closing && isSelfClosing)
1042 						ret.push(token.closing.val);
1043 				}
1044 				else
1045 					ret.push(token.val);
1046 			}
1047 
1048 			return ret.join('');
1049 		};
