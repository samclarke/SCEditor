 75 	/**
 76 	 * <p>Replaces any params in a template with the passed params.</p>
 77 	 *
 78 	 * <p>If createHTML is passed it will use jQuery to create the HTML. The
 79 	 * same as doing: $(editor.tmpl("html", {params...}));</p>
 80 	 *
 81 	 * @param {string} templateName
 82 	 * @param {Object} params
 83 	 * @param {Boolean} createHTML
 84 	 * @private
 85 	 */
 86 	var _tmpl = function(name, params, createHTML) {
 87 		var template = _templates[name];
 88 
 89 		$.each(params, function(name, val) {
 90 			template = template.replace(new RegExp('\\{' + name + '\\}', 'g'), val);
 91 		});
 92 
 93 		if(createHTML)
 94 			template = $(template);
 95 
 96 		return template;
 97 	};

583 		/**
584 		 * Creates the resizer.
585 		 * @private
586 		 */
587 		initResize = function () {
588 			var	minHeight, maxHeight, minWidth, maxWidth, mouseMoveFunc, mouseUpFunc,
589 				$grip       = $('<div class="sceditor-grip" />'),
590 				// cover is used to cover the editor iframe so document still gets mouse move events
591 				$cover      = $('<div class="sceditor-resize-cover" />'),
592 				startX      = 0,
593 				startY      = 0,
594 				startWidth  = 0,
595 				startHeight = 0,
596 				origWidth   = $editorContainer.width(),
597 				origHeight  = $editorContainer.height(),
598 				dragging    = false,
599 				rtl         = base.rtl();
600 
601 			minHeight = base.opts.resizeMinHeight || origHeight / 1.5;
602 			maxHeight = base.opts.resizeMaxHeight || origHeight * 2.5;
603 			minWidth  = base.opts.resizeMinWidth  || origWidth  / 1.25;
604 			maxWidth  = base.opts.resizeMaxWidth  || origWidth  * 1.25;
605 
606 			mouseMoveFunc = function (e) {
607 				// iOS must use window.event
608 				if(e.type === 'touchmove')
609 					e = window.event;
610 
611 				var	newHeight = startHeight + (e.pageY - startY),
612 					newWidth  = rtl ? startWidth - (e.pageX - startX) : startWidth + (e.pageX - startX);
613 
614 				if(maxWidth > 0 && newWidth > maxWidth)
615 					newWidth = maxWidth;
616 
617 				if(maxHeight > 0 && newHeight > maxHeight)
618 					newHeight = maxHeight;
619 
620 				if(base.opts.resizeWidth && newWidth >= minWidth && (maxWidth < 0 || newWidth <= maxWidth))
621 					base.width(newWidth);
622 
623 				if(base.opts.resizeHeight && newHeight >= minHeight && (maxHeight < 0 || newHeight <= maxHeight))
624 					base.height(newHeight);
625 
626 				e.preventDefault();
627 			};
628 
629 			mouseUpFunc = function (e) {
630 				if(!dragging)
631 					return;
632 
633 				dragging = false;
634 
635 				$cover.hide();
636 				$editorContainer.removeClass('resizing');
637 				$(document).unbind('touchmove mousemove', mouseMoveFunc);
638 				$(document).unbind('touchend mouseup', mouseUpFunc);
639 
640 				e.preventDefault();
641 			};
642 
643 			$editorContainer.append($grip);
644 			$editorContainer.append($cover.hide());
645 
646 			$grip.bind('touchstart mousedown', function (e) {
647 				// iOS must use window.event
648 				if(e.type === 'touchstart')
649 					e = window.event;
650 
651 				startX      = e.pageX;
652 				startY      = e.pageY;
653 				startWidth  = $editorContainer.width();
654 				startHeight = $editorContainer.height();
655 				dragging    = true;
656 
657 				$editorContainer.addClass('resizing');
658 				$cover.show();
659 				$(document).bind('touchmove mousemove', mouseMoveFunc);
660 				$(document).bind('touchend mouseup', mouseUpFunc);
661 
662 				e.preventDefault();
663 			});
664 		};
